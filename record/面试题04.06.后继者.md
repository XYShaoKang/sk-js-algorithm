## 解法 1: 中序遍历

```js
var inorderSuccessor = function (root, p) {
  let nodes = []
  const dfs = node => {
    if (node.left) dfs(node.left)
    nodes.push(node)
    if (node.right) dfs(node.right)
  }
  dfs(root)
  const n = nodes.length
  for (let i = 0; i < n; i++) {
    if (nodes[i] === p) return i === n - 1 ? null : nodes[i + 1]
  }
}
```
