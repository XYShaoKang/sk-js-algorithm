## 解法 1: 前后同时迭代

```ts
interface ListNode {
  val: number
  next: ListNode | null
  pre?: ListNode | null
}
function isPalindrome(head: ListNode | null): boolean {
  let root = head
  while (head && head.next) {
    head.next.pre = head
    head = head.next
  }
  let res = true
  while (root) {
    if (root.val !== head?.val) {
      res = false
    }
    root = root.next
    head = head?.pre!
    delete head?.next!.pre
  }
  return res
}
```

## 解法 2: 使用一个外部数组

```ts
function isPalindrome(head: ListNode | null): boolean {
  const cache: ListNode[] = []
  while (head) {
    cache.push(head)
    head = head.next
  }
  const n = cache.length
  for (let i = 0; i < n / 2; i++) {
    if (cache[i].val !== cache[n - i - 1].val) {
      return false
    }
  }
  return true
}
```
