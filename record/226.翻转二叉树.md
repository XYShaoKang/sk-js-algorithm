## 解法 1: 递归

时间复杂度: `O(n)`
空间复杂度: `O(n)`

```ts
function invertTree(root: TreeNode | null): TreeNode | null {
  if (!root) {
    return root
  }
  invertTree(root.left)
  invertTree(root.right)
  ;[root.left, root.right] = [root.right, root.left]
  return root
}
```

```ts
function invertTree(root: TreeNode | null): TreeNode | null {
  if (root)
    [root.left, root.right] = [invertTree(root.right), invertTree(root.left)]

  return root
}
```
