## 解法 1: 哈希表

时间复杂度: `O(n)`
空间复杂度: `O(n)`

```ts
function hasCycle(head: ListNode | null): boolean {
  const cache = new Map()
  let cur = head
  cache.set(cur, 1)
  while (cur) {
    if (cache.has(cur.next)) {
      return true
    }
    cache.set(cur.next, 1)
    cur = cur.next
  }
  return false
}
```

## 解法 2: 快慢指针

时间复杂度: `O(n)`
空间复杂度: `O(1)`

```ts
function hasCycle(head: ListNode | null): boolean {
  if (!head) {
    return false
  }

  let slow: ListNode | null = head
  let fast: ListNode | null = head
  while (fast !== null) {
    slow = slow?.next ?? null
    fast = fast.next?.next ?? null
    if (fast && slow === fast) {
      return true
    }
  }
  return false
}
```
