## 解法 1: 动态规划

```ts
function generate(numRows: number): number[][] {
  const dp: number[][] = [[1]]
  for (let i = 1; i < numRows; i++) {
    dp[i] = []
    for (let j = 0; j <= i; j++) {
      dp[i][j] = (dp[i - 1][j] ?? 0) + (dp[i - 1][j - 1] ?? 0)
    }
  }
  return dp
}
```

## Case

```ts
test.each([

    input: { numRows: 5 },
    output: [[1], [1, 1], [1, 2, 1], [1, 3, 3, 1], [1, 4, 6, 4, 1]],

  { input: { numRows: 1 }, output: [[1]] },

  expect(generate(numRows)).toEqual(output)
})
```
