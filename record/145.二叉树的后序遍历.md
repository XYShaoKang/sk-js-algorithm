## 解法 1: Morris

```ts
function postorderTraversal(root: TreeNode | null): number[] {
  let res: number[] = []
  while (root) {
    if (root.right) {
      let pre = root.right
      while (pre.left && pre.left !== root) {
        pre = pre.left!
      }
      if (!pre.left) {
        pre.left = root
        res.push(root.val)
        root = root.right
      } else {
        root = root.left
      }
    } else {
      res.push(root.val)
      root = root.left
    }
  }
  return res.reverse()
}
```
