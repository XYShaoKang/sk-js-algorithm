## 解法 1: 双指针

- 时间复杂度: $O(n)$
- 空间复杂度: $O(1)$

```ts
function moveZeroes(nums: number[]): void {
  let j = 0
  for (let i = 0; i < nums.length; i++) {
    const num = nums[i]
    if (i !== j) {
      nums[i] = nums[j]
      nums[j] = num
    }
    if (num !== 0) {
      j++
    }
  }
}
```

## 解法 2: 滚雪球

- 时间复杂度: $O(n)$
- 空间复杂度: $O(1)$

https://leetcode.com/problems/move-zeroes/discuss/172432

```ts
function moveZeroes(nums: number[]): void {
  let offset = 0
  for (let i = 0; i < nums.length; i++) {
    const num = nums[i]
    if (num !== 0) {
      nums[i] = 0
      nums[i - offset] = num
    } else {
      offset++
    }
  }
}
```
