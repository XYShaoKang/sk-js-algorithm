## 解法 1: 对比二进制字符串

```ts
function hammingDistance(x: number, y: number): number {
  if (x > y) [x, y] = [y, x]

  let [xstring, ystring] = [x, y].map((n) => n.toString(2))
  xstring = xstring.padStart(ystring.length, '0')
  let res = 0
  for (let i = 0; i < xstring.length; i++) {
    if (xstring[i] !== ystring[i]) res++
  }
  return res
}
```
