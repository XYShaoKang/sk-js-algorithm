## 解法 1: 暴力枚举

- 时间复杂度: $O(n^2)$
- 空间复杂度: $O(1)$

```ts
function twoSum(nums: number[], target: number): number[] {
  for (let i = 0; i < nums.length; i++) {
    for (let j = i + 1; j < nums.length; j++) {
      if (nums[i] + nums[j] === target) {
        return [i, j]
      }
    }
  }
  return []
}
```

## 解法 2: 哈希表

- 时间复杂度: $O(n)$
- 空间复杂度: $O(n)$

```ts
function twoSum(nums: number[], target: number): number[] {
  const cache = new Map()
  for (let i = 0; i < nums.length; i++) {
    const num = nums[i]
    const needNum = target - num
    if (cache.has(needNum)) {
      return [cache.get(needNum), i]
    } else {
      cache.set(num, i)
    }
  }
  return []
}
```
