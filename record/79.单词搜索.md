## 解法 1: 回溯(超时)

```ts
function exist(board: string[][], word: string): boolean {
  // 方向数组
  const direction = [
    [0, 1],
    [0, -1],
    [1, 0],
    [-1, 0],
  ]
  // 记录是否匹配
  let res = false
  const helper = (x = 0, y = 0, cache = new Set<string>()) => {
    // 边界处理: 是否超出网格下标,是否已经反问过,以及是否已经匹配成功
    if (
      x === board[0].length ||
      y === board.length ||
      x < 0 ||
      y < 0 ||
      cache.has(`${x},${y}`) ||
      res
    ) {
      return
    }

    if (board[y][x] === word[cache.size]) {
      if (cache.size === word.length - 1) {
        // 匹配到最后一个字符相等时,表示可以匹配当前单词
        res = true
        return
      }

      cache.add(`${x},${y}`)
      for (const [dirx, diry] of direction) {
        // 进入下一层,分别去尝试上下左右是否匹配
        helper(x + dirx, y + diry, cache)
      }
      cache.delete(`${x},${y}`)
    }
    if (cache.size === 0) {
      // 匹配第一个字符,网格中每个字符都匹配一下
      for (const [dirx, diry] of [
        [1, 0],
        [0, 1],
      ]) {
        helper(x + dirx, y + diry, cache)
      }
    }
  }
  helper()
  return res
}
```

### 优化

```ts
function exist(board: string[][], word: string): boolean {
  // 方向数组
  const direction = [
    [0, 1],
    [0, -1],
    [1, 0],
    [-1, 0],
  ]

  const cache = new Array(board.length)
    .fill(0)
    .map(() => new Array(board[0].length).fill(false))

  const helper = (x = 0, y = 0, i = 0) => {
    // 边界处理: 是否超出网格下标,是否已经访问过
    if (
      x === board[0].length ||
      y === board.length ||
      x < 0 ||
      y < 0 ||
      cache[y][x]
    ) {
      return false
    }

    if (board[y][x] === word[i]) {
      // 最后一个字符匹配成功,则表示能搜索到想要的单词
      if (i === word.length - 1) return true

      cache[y][x] = true
      for (const [dirx, diry] of direction) {
        if (helper(x + dirx, y + diry, i + 1)) return true
      }
      cache[y][x] = false
    }
  }

  for (let i = 0; i < board.length; i++) {
    for (let j = 0; j < board[i].length; j++) {
      if (helper(j, i)) return true
    }
  }
  return false
}
```
