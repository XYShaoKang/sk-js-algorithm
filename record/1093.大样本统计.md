```ts
function sampleStats(count: number[]): number[] {
  const arr = count.map((num, i) => [num, i]).filter(([num]) => num)
  let min = arr[0][1],
    max = arr[arr.length - 1][1],
    sum = 0,
    cnt = 0,
    mode = -1,
    cnt1 = 0
  for (let [num, i] of arr) {
    sum += num * i
    cnt += num
    if (mode === -1 || num > cnt1) {
      mode = i
      cnt1 = num
    }
  }
  let mean = sum / cnt
  const find = (t: number) => {
    let cnt = 0
    for (let [num, i] of arr) {
      cnt += num
      if (cnt >= t) return i
    }
  }

  let median: number
  if (cnt & 1) {
    median = find(Math.ceil(cnt / 2))
  } else {
    median = (find(cnt / 2) + find(cnt / 2 + 1)) / 2
  }
  return [min, max, mean, median, mode]
}
```
