## 解法 1: 辅助数组 + 快速排序

```ts
function sortList(head: ListNode | null): ListNode | null {
  if (!head) return null

  let arr: ListNode[] = []
  while (head) {
    arr.push(head)
    head = head.next
  }
  const helper = (start = 0, end = arr.length - 1) => {
    if (start === end) return
    let cur = start,
      mid = arr[start].val
    for (let i = start; i <= end; i++) {
      if (mid > arr[i].val) {
        ;[arr[cur], arr[i]] = [arr[i], arr[cur]]
        cur++
      }
    }
    if (cur === start) cur++
    helper(start, cur - 1)
    helper(cur, end)
  }
  helper()
  const root = arr[0]
  for (let i = 0; i < arr.length; i++) {
    arr[i].next = arr[i + 1] ?? null
  }
  return root
}
```

## 解法 2: 辅助数组 + 归并排序

```ts
function sortList(head: ListNode | null): ListNode | null {
  if (!head) return null

  let arr: ListNode[] = []
  while (head) {
    arr.push(head)
    head = head.next
  }
  const helper = (start = 0, end: number): ListNode => {
    if (start === end) {
      arr[start].next = null
      return arr[start]
    }

    const mid = (start + end) >>> 1
    let left: ListNode | null = helper(start, mid)
    let right: ListNode | null = helper(mid + 1, end)

    // 这里一定要是小于等于,不然如果有相等的元素,则 root 不会记录 left
    const root = left.val <= right.val ? left : right
    while (left && right) {
      if (left.val > right.val) [left, right] = [right, left]

      const next: ListNode | null = left.next
      if (!left.next || left.next.val > right.val) left.next = right
      left = next
    }
    return root
  }
  return helper(0, arr.length - 1)
}
```

## 解法 3: 双指针 + 归并排序

使用双指针来确定中间节点

```ts
function sortList(head: ListNode | null): ListNode | null {
  if (!head) return null

  function merge(node1: ListNode | null, node2: ListNode | null) {
    if (!node1) return node2
    if (!node2) return node1

    const root = node1.val <= node2.val ? node1 : node2
    while (node1) {
      if (node1.val > node2.val) [node1, node2] = [node2, node1]
      const next = node1.next
      if (!node1.next || node1.next.val > node2.val) node1.next = node2
      node1 = next!
    }
    return root
  }

  const helper = (root: ListNode, tail: ListNode | null): ListNode => {
    if (!root) return root

    if (root.next === tail) {
      root.next = null
      return root
    }

    let slow = root,
      fast = root.next
    while (fast !== tail) {
      slow = slow.next!
      fast = fast?.next!
      if (fast !== tail) fast = fast.next
    }

    let node1 = helper(root, slow),
      node2 = helper(slow, tail)
    return merge(node1, node2)
  }
  return helper(head, null)
}
```

## 解法 4: 归并排序 - 自底向上

```ts
function sortList(head: ListNode | null): ListNode | null {
  if (!head) return null

  function merge(node1: ListNode | null, node2: ListNode | null) {
    if (!node1) return node2
    if (!node2) return node1

    const root = node1.val <= node2.val ? node1 : node2
    while (node1) {
      if (node1.val > node2.val) [node1, node2] = [node2, node1]
      const next = node1.next
      if (!node1.next || node1.next.val > node2.val) node1.next = node2
      node1 = next!
    }
    return root
  }

  function split(root: ListNode | null, step: number): ListNode | null {
    if (!root) return root

    while (step - 1 && root.next) {
      root = root.next!
      step--
    }
    let rignt = root.next
    root.next = null
    return rignt
  }

  let len = 0,
    root = head
  while (head) {
    len++
    head = head.next
  }
  let res: ListNode = { val: 0, next: root }
  for (let i = 1; i <= len; i <<= 1) {
    let pre = res
    let cur: ListNode | null = res.next
    while (cur) {
      let h1 = cur
      let h2 = split(cur, i)
      cur = split(h2, i)
      pre.next = merge(h1, h2)
      while (pre.next) {
        pre = pre.next
      }
    }
  }

  return res.next
}
```
