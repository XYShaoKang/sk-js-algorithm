## 解法 1: 回溯

```ts
function findWords(board: string[][], words: string[]): string[] {
  const dir = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ]
  const [m, n] = [board.length, board[0].length]
  const cache = new Array(m).fill(0).map(() => new Array(n).fill(false))
  const search = (i = 0, j = 0, k = 0, word = '') => {
    if (i < 0 || j < 0 || i >= m || j >= n || cache[i][j]) return false
    if (board[i][j] === word[k]) {
      if (k === word.length - 1) return true

      cache[i][j] = true
      let res = false
      for (const [x, y] of dir) {
        if (search(i + x, j + y, k + 1, word)) res = true
      }
      cache[i][j] = false
      return res
    }
  }
  const res = new Set<string>()
  for (let i = 0; i < board.length; i++) {
    for (let j = 0; j < board[i].length; j++) {
      for (const word of words) {
        if (!res.has(word) && search(i, j, 0, word)) {
          res.add(word)
        }
      }
    }
  }

  return [...res]
}
```

## 解法 2: 使用字典树

```ts
type TrieNode = {
  [key: string]: TrieNode
} & { isEnd?: boolean; word?: string }

function findWords(board: string[][], words: string[]): string[] {
  const dir = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ]
  const [m, n] = [board.length, board[0].length]
  const cache = new Array(m).fill(0).map(() => new Array(n).fill(false))

  // 初始化字典树
  const trie: TrieNode = {}
  for (const word of words) {
    let node = trie
    for (const c of word) {
      node = node[c] || (node[c] = {})
    }
    node.isEnd = true
    node.word = word
  }

  const res = new Set<string>()
  const search = (i = 0, j = 0, node: TrieNode) => {
    // 边界处理: 超出网格坐标,已经访问过
    if (i < 0 || j < 0 || i >= m || j >= n || cache[i][j]) return false
    const next = node[board[i][j]]
    if (next) {
      if (next.isEnd) {
        console.log(next.word)
        res.add(next.word!)

        // 从字典树中删除已经匹配的单词
        delete next.isEnd
        delete next.word
        if (Object.keys(next).length === 0) {
          delete node[board[i][j]]
          return
        }
      }

      cache[i][j] = true
      for (const [x, y] of dir) {
        search(i + x, j + y, next)
      }
      cache[i][j] = false

      // 从字典树中删除已经匹配的单词
      if (Object.keys(next).length === 0) delete node[board[i][j]]
    }
  }
  for (let i = 0; i < board.length; i++) {
    for (let j = 0; j < board[i].length; j++) {
      search(i, j, trie)
    }
  }

  return [...res]
}
```

test.each([
  {
    input: {

        ['i', 'f', 'l', 'v'],

      board: [
        ['a', 'b'],
