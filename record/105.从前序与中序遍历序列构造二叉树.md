# 105.从前序与中序遍历序列构造二叉树

```ts
/*
 * @lc app=leetcode.cn id=105 lang=typescript
 *
 * [105] 从前序与中序遍历序列构造二叉树
 */

// @lc code=start
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     val: number
 *     left: TreeNode | null
 *     right: TreeNode | null
 *     constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.left = (left===undefined ? null : left)
 *         this.right = (right===undefined ? null : right)
 *     }
 * }
 */
interface TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
}
function buildTree(preorder: number[], inorder: number[]): TreeNode | null {}
// @lc code=end
```

## 解法 1: 递归

```ts
function buildTree(preorder: number[], inorder: number[]): TreeNode | null {
  if (preorder.length === 0) {
    return null
  }
  const left = inorder.indexOf(preorder[0])

  return {
    val: preorder[0],
    left: buildTree(preorder.slice(1, left + 1), inorder.slice(0, left)),
    right: buildTree(
      preorder.slice(left + 1, preorder.length),
      inorder.slice(left + 1, inorder.length),
    ),
  }
}
```

## 解法 2: 递归

- 时间复杂度: <!-- $O(n)$ --> <img style="transform: translateY(0.1em); background: white;" src="./svg/o-n.svg" alt="O(n)">
- 空间复杂度: <!-- $O(n)$ --> <img style="transform: translateY(0.1em); background: white;" src="./svg/o-n.svg" alt="O(n)">

```ts
function buildTree(preorder: number[], inorder: number[]): TreeNode | null {
  if (preorder.length === 0) {
    return null
  }

  let cache: { [key: number]: number } = {}
  const lenght = inorder.length
  for (let i = 0; i < lenght; i++) {
    cache[inorder[i]] = i
  }

  const helper = (
    preorder: number[],
    inorder: number[],
    pstart: number = 0,
    pend: number = lenght,
    istart: number = 0,
    iend: number = lenght,
  ): TreeNode | null => {
    if (pstart === pend) {
      return null
    }
    const rootIndex = cache[preorder[pstart]] - istart
    return {
      val: preorder[pstart],
      left: helper(
        preorder,
        inorder,
        pstart + 1,
        pstart + rootIndex + 1,
        istart,
        istart + rootIndex,
      ),
      right: helper(
        preorder,
        inorder,
        pstart + rootIndex + 1,
        pend,
        istart + rootIndex + 1,
        iend,
      ),
    }
  }
  return helper(preorder, inorder)
}
```

## 解法 3: 迭代

```ts
function buildTree(preorder: number[], inorder: number[]): TreeNode | null {
  if (preorder.length === 0) {
    return null
  }
  const root = { val: preorder[0], left: null, right: null }
  const stack: Array<TreeNode> = [root]
  let j = 0
  for (let i = 1; i < preorder.length; i++) {
    const cur = preorder[i]
    let top = stack[stack.length - 1]

    if (top.val !== inorder[j]) {
      top.left = { val: cur, left: null, right: null }
      stack.push(top.left)
    } else {
      while (stack.length && stack[stack.length - 1].val === inorder[j]) {
        j++
        top = stack.pop()!
      }

      top.right = { val: cur, left: null, right: null }
      stack.push(top.right)
    }
  }
  return root
}
```
