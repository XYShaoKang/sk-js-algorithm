## 解法 1: 递归 前序遍历

时间复杂度: `O(n)`
空间复杂度: `O(n)`

```ts
function isValidBST(
  root: TreeNode | null,
  left: number = -Infinity,
  right: number = Infinity,
): boolean {
  if (!root) return true

  if (root.val <= left || root.val >= right) return false

  return (
    isValidBST(root.left, left, root.val) &&
    isValidBST(root.right, root.val, right)
  )
}
```

## 解法 1: 递归 中序遍历

时间复杂度: `O(n)`
空间复杂度: `O(n)`

```ts
function isValidBST(
  root: TreeNode | null,
  pre: { value: number } = { value: -Infinity },
): boolean {
  if (!root) return true

  if (!isValidBST(root.left, pre)) return false

  if (root.val <= pre.value) return false

  pre.value = root.val

  return isValidBST(root.right, pre)
}
```

## 解法 3: 迭代 中序遍历

时间复杂度: `O(n)`
空间复杂度: `O(n)`

```ts
function isValidBST(root: TreeNode | null): boolean {
  if (!root) {
    return true
  }

  const stack: Array<TreeNode> = []
  let pre = -Infinity
  while (root || stack.length) {
    while (root) {
      stack.push(root)
      root = root.left
    }

    let tmp = stack.pop()!
    if (tmp.val <= pre) {
      return false
    }
    pre = tmp.val

    root = tmp.right
  }
  return true
}
```
