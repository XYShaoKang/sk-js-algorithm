## 解法 1: 动态规划

```ts
function getRow(rowIndex: number): number[] {
  const dp = [1]
  for (let i = 1; i < rowIndex + 1; i++) {
    for (let j = i; j >= 0; j--) {
      dp[j] = (dp[j] ?? 0) + (dp[j - 1] ?? 0)
    }
  }
  return dp
}
```

test.each([
