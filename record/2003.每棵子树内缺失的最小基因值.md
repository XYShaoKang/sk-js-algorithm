## 解法 1: 深度优先遍历

```ts
function smallestMissingValueSubtree(
  parents: number[],
  nums: number[]
): number[] {
  const map: { [k: number]: number[] } = {}

  for (let i = 0; i < parents.length; i++) {
    const parent = parents[i]
    if (!map[parent]) map[parent] = []
    map[parent].push(i)
  }
  let res: number[] = []
  const dfs = (parent: number): [number, Set<number>] => {
    const children = map[parent] ?? []
    let min = 1,
      set = new Set<number>([nums[parent]])
    for (const child of children) {
      let [m, s] = dfs(child)
      min = Math.max(min, m)
      if (s.size > set.size) [s, set] = [set, s]

      for (const num of s) {
        set.add(num)
      }
    }

    while (set.has(min)) {
      min++
    }
    res[parent] = min
    return [min, set]
  }
  dfs(0)
  return res
}
```

## Case

```ts
test.each([
  {
    input: { parents: [-1, 0, 0, 2], nums: [1, 2, 3, 4] },
    output: [5, 1, 1, 1],
  },
  {
    input: { parents: [-1, 0, 1, 0, 3, 3], nums: [5, 4, 6, 2, 1, 3] },
    output: [7, 1, 1, 4, 2, 1],
  },
  {
    input: { parents: [-1, 2, 3, 0, 2, 4, 1], nums: [2, 3, 4, 5, 6, 7, 8] },
    output: [1, 1, 1, 1, 1, 1, 1],
  },
])(
  'input: parents = $input.parents, nums = $input.nums',
  ({ input: { parents, nums }, output }) => {
    expect(smallestMissingValueSubtree(parents, nums)).toEqual(output)
  }
)
```
